# apt packages installed into the kata-guest-base rootfs (passed via
# `steep build -p`). One package per line. Lines starting with `#` are
# ignored. build.sh reads this file and turns it into a comma-separated
# --package= list.
#
# The minimum for a kata guest that runs systemd as the in-VM init,
# accepts cloud-init user-data via the NoCloud datasource, and hosts
# the in-guest ratls-mesh + attestation-service services.
#
# Kernel: distro kernel is NOT used. steep ships its hardened guest
# kernel built from kernel/container.config + the upstream kata-friendly
# guest options.

# Core init + service supervision. mkosi's base image already includes
# systemd, but we depend on its presence so service files in extra/
# get processed by `systemctl preset-all` during the build.
systemd
systemd-sysv

# cloud-init reads per-pod user-data injected by kata as a NoCloud
# datasource (9p or vfat ISO). Without this, the C8S_* env vars in
# /etc/c8s/cloudinit.env never arrive.
cloud-init

# Standard C library + DNS / TLS roots. Mkosi base usually pulls these
# in but be explicit so a refactor of the base image cannot drop them.
libc-bin
ca-certificates

# Networking utilities — operator-facing diagnostics inside the guest.
# Also needed by ratls-mesh's iptables-redirect setup (uses iptables
# under the hood; the binary statically links libc but invokes
# /sbin/iptables for nft compatibility on the guest kernel).
iproute2
iputils-ping
iptables

# ipset — ratls-mesh tracks the pod's peer CIDRs as an ipset for fast
# match in the REDIRECT chain.
ipset

# Diagnostics on a wedged in-guest service. Removable if image size
# becomes a constraint; cloud-init pulls procps anyway.
procps

# Curl is invoked by cloudinit-env.sh's optional fallback when the
# NoCloud datasource is empty. Tiny; keep it.
curl
